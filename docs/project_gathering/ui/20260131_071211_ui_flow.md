UI Flow (CLI)

START
  |
  v
[kmi --help] -> show help and exit
  |
  +--> [kmi --rotate] ----> load config/auths/state -> health -> rotate -> dashboard
  |
  +--> [kmi rotate auto] -> enable auto-rotation -> optional e2e -> exit
  |
  +--> [kmi rotate off]  -> disable auto-rotation -> exit
  |
  +--> [kmi --all|--health] -> load config/auths/state -> health dashboard
  |
  +--> [kmi --current] -> read ~/.kimi/config.toml -> health dashboard
  |
  +--> [kmi proxy] -> start FastAPI proxy
  |
  +--> [kmi trace] -> live TUI (trace.jsonl)
  |
  +--> [kmi e2e] -> ensure proxy -> send batches -> confidence summary
  |
  +--> [kmi status] -> show active/rotation indices
  v
END
